    <script>
        const audio = document.getElementById('global-audio');
        const landing = document.getElementById('landing-page');
        const reel = document.getElementById('reel');
        const pages = document.querySelectorAll('.reel-page');
        let currentPageIndex = -1;

        function enterExperience() {
            landing.classList.add('fade-out');
            reel.style.display = 'block';
            document.body.style.overflow = 'auto';
            setTimeout(() => { handleScroll(); }, 800);
        }

        function handleScroll() {
            const scrollPos = reel.scrollTop;
            const windowHeight = window.innerHeight;
            const newIndex = Math.round(scrollPos / windowHeight);
            if (newIndex !== currentPageIndex) {
                currentPageIndex = newIndex;
                updateMusic(currentPageIndex);
            }
        }

        function updateMusic(index) {
            const activePage = pages[index];
            if (!activePage) return;
            const songSrc = activePage.getAttribute('data-song');
            if (songSrc) {
                if (audio.src.split('/').pop() !== songSrc) {
                    audio.pause();
                    audio.src = songSrc;
                    audio.load();
                    audio.play().catch(e => console.log("Play deferred:", e));
                }
            } else {
                audio.pause();
                audio.currentTime = 0;
            }
        }
        reel.addEventListener('scroll', handleScroll);
    </script>
